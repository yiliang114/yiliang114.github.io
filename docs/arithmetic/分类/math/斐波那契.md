# 斐波那契

## Code

```js
// 1,1,2,3,5,8...

// 时间复杂度 O(n) 空间复杂度 O(1)
function fib(n) {
  if (n < 3) return 1;
  let f = 0,
    s = 1;
  while (--n) {
    // s += f; // f + s
    // f = s - f; // 旧 s 值
    [f, s] = [s, f + s];
  }
  return s;
}

console.log(fib(1));
console.log(fib(3));
console.log(fib(5));
console.log(fib(6));
console.log(fib(10));
```

```js
/// solution

function fib(n) {
  switch (n) {
    case 0:
      return 0;
    case 1:
      return 1;
    default:
      return fib(n - 1) + fib(n - 2);
  }
}

/// tests

import { test } from 'ava';

test(t => t.is(fib(0), 0));
test(t => t.is(fib(1), 1));
test(t => t.is(fib(10), 55));
test(t => t.is(fib(20), 6765));
```
